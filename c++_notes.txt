// variadic macro
// log with filename, line number source location

#define LOG(fmt, ...) printf("%s:%d: " fmt "\n", __FILE__, __LINE__, ##__VA_ARGS__)
LOG();
LOG("");
LOG("%s, %d, %f", "hello", 123, 3.14);

$ ./src/main/main
main.c:10:
main.c:11:
main.c:12: hello, 123, 3.140000
$
